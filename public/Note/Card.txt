import React, { useState } from "react";

const ProductItem = () => {
  let [state, setState] = useState({
    product: {
      sno: "AAOO1",
      name: "MI Watch",
      image:
        "http://cdn.shopify.com/s/files/1/0997/6284/products/Side04_grande.png?v=1655378358",
      price: 1500,
      qty: 2,
    },
  });
  let { product } = state;
  const Increement = () => {
    setState(() => ({
      product: {
        ...state.product,
        qty: state.product.qty + 1,
      },
    }));
  };
  const Decreement = () => {
    setState(() => ({
      product: {
        ...state.product,
        qty: state.product.qty - 1 > 0 ? state.product.qty - 1 : 1,
      },
    }));
  };
  return (
    <React.Fragment>
      <div className="container mt-3">
        <div className="row">
          <div className="col">
            <p className="h3 text-success">ProductItem</p>
            <p>
              Lorem, ipsum dolor sit amet consectetur adipisicing elit.
              Perspiciatis, illum ipsam. Error ad magni maxime magnam.
              Consectetur impedit debitis consequuntur natus est fuga voluptas
              esse ullam dicta nihil, temporibus accusantium?
            </p>
          </div>
        </div>
      </div>
      <div className="container mt-3">
        <div className="row">
          <div className="col">
            <table className="table table-striped text-center table-hover">
              <thead className="bg-dark text-white">
                <tr>
                  <th>SNO</th>
                  <th>Image</th>
                  <th>Name</th>
                  <th>Price</th>
                  <th>Qty</th>
                  <th>Total</th>
                </tr>
              </thead>
              <tbody>
                <tr>
                  <td>{product.sno}</td>
                  <td>
                    <img
                      src={product.image}
                      alt={product.name}
                      width={50}
                      height={50}
                    />
                  </td>
                  <td>{product.name}</td>
                  <td>{product.price}</td>
                  <td>
                    <i
                      onClick={Decreement}
                      className="fa fa-minus-square mx-1"
                    />

                    {product.qty}
                    <i
                      onClick={Increement}
                      className="fa fa-plus-square mx-1"
                    />
                  </td>
                  <td>{product.qty * product.price}</td>
                </tr>
              </tbody>
            </table>
          </div>
        </div>
      </div>
    </React.Fragment>
  );
};

export default ProductItem;
